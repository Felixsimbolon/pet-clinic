{% block extra_js %}
<script>
    document.addEventListener('DOMContentLoaded', function () {
        // Handle update stock button clicks
        document.querySelectorAll('.btn-update-stock').forEach(function(btn) {
            btn.addEventListener('click', function() {
                console.log('Update stock button clicked'); // Debug log
                
                // Get data from attributes
                const id = this.dataset.id;
                const name = this.dataset.name;
                const stock = this.dataset.stock;
                const url = this.dataset.url;
                
                console.log('Data:', {id, name, stock, url}); // Debug log
                
                // Fill form and title
                document.getElementById('update_stock_title').textContent = name + ' (' + id + ')';
                document.getElementById('update_stock_id_vaksin').value = id;
                document.getElementById('update_stock_value').value = stock;
                document.getElementById('updateStockForm').action = url;
                
                console.log('Form action set to:', url); // Debug log
            });
        });
        
        // Handle update vaccine button clicks
        document.querySelectorAll('.btn-update-vaccine').forEach(function(button) {
            button.addEventListener('click', function() {
                console.log('Update vaccine button clicked'); // Debug log
                
                var id = this.getAttribute('data-id');
                var name = this.getAttribute('data-name');
                var price = this.getAttribute('data-price');
                
                // Fill form fields
                document.getElementById('update_id_vaksin').value = id;
                document.getElementById('update_nama_vaksin').value = name;
                document.getElementById('update_harga').value = price.replace(/[^\d]/g, ''); // Remove non-numeric chars
                
                // Update form action
                document.getElementById('updateVaccineForm').action = '{% url "manajemen_vaksinasi:edit_vaccine" "0" %}'.replace('0', id);
            });
        });

        // Search functionality
        const searchInput = document.getElementById('searchInput');
        const clearSearchBtn = document.getElementById('clearSearch');
        const vaccineRows = document.querySelectorAll('.vaccine-row');
        const noResultsMessage = document.getElementById('noResultsMessage');
        
        searchInput.addEventListener('input', function() {
            const searchTerm = this.value.toLowerCase();
            let matchFound = false;
            
            vaccineRows.forEach(function(row) {
                const vaccineId = row.querySelector('.vaccine-id').textContent.toLowerCase();
                const vaccineName = row.querySelector('.vaccine-name').textContent.toLowerCase();
                
                if (vaccineId.includes(searchTerm) || vaccineName.includes(searchTerm)) {
                    row.classList.remove('d-none');
                    matchFound = true;
                } else {
                    row.classList.add('d-none');
                }
            });
            
            // Show/hide no results message
            if (matchFound) {
                noResultsMessage.classList.add('d-none');
            } else {
                noResultsMessage.classList.remove('d-none');
            }
        });
        
        // Clear search function
        clearSearchBtn.addEventListener('click', function() {
            searchInput.value = '';
            vaccineRows.forEach(function(row) {
                row.classList.remove('d-none');
            });
            noResultsMessage.classList.add('d-none');
        });
    });
</script>
